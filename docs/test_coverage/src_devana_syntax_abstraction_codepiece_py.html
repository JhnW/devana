70 @property 

71 def text(self) -> str: 

72 """Raw text of code, from begin to end.""" 

73 if self._text is None: 

74 try: 

75 # standard file 

76 with open(self.file, "rb") as f: 

77 if self._cursor is None: 

78 lines = f.read().split(b'\n') 

79 row_start = self.begin.row-1 

80 row_end = self.end.row-1 

81 col_start = self.begin.col-1 

82 col_end = self.end.col 

83 text = b"" 

84 if len(lines) < row_start or len(lines) < row_end: 

85 raise ValueError("Code begin and end extend file size.") 

86 for index, line in enumerate(lines): 

87 if row_start == index and row_end == index: 

88 text = line[col_start:col_end+1] 

89 if len(lines) != index+1: 

90 text += b'\n' 

91 return str(text) 

92 elif row_start == index: 

93 text += line[col_start:]+b'\n' 

94 elif row_end == index: 

95 text += line[:col_end] 

96 elif row_start <= index <= row_end: 

97 text += line+b'\n' 

98 elif row_end < index: 

99 break 

100 return str(text) 

101 else: 

102 begin = self._cursor.extent.start.offset 

103 end = self._cursor.extent.end.offset 

104 f.seek(begin) 

105 return self._remove_base_indent(f.read(end - begin).decode()) 

106 except IOError: 

107 # clang in memory file (special dragon case) - do not look 

108 return self._cursor.spelling 

109 return self._text

116 def _remove_base_indent(text: Optional[str]) -> Optional[str]: 

117 split_text = text.split("\n") 

118 last_line = split_text[-1] 

119 indent_len = len(last_line) - len(last_line.lstrip()) 

120 prefix = "" 

121 for i in range(indent_len): 

122 prefix += " " 

123 result = "" 

124 for line in split_text: 

125 result += line[line.startswith(prefix) and len(prefix):] 

126 result += "\n" 

127 return result[:-1]
